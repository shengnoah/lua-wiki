function n_serialize( o )
    if type(o) == &#34;number&#34; then
        io.write(o)
    elseif type(o) == &#34;string&#34; then
        io.write(string.format('%q',o))
    elseif type(o) == &#34;table&#34; then
        io.write(&#34;{n&#34;)
        for k,v in pairs(o) do
            io.write(&#34; &#34;, k, &#34;=&#34;)
            – 递归调用
            n_serialize(v)
            io.write(&#34;,n&#34;)
        end
        io.write(&#34;}n&#34;)
    else
        print('b')
    end
end
