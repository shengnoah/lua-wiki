–[[
–加上travelled_tables是为了防止互相引用，用它来存储已经设置过proxy的table的映射，比如
local t = {a = 1, b = 2}
t.c = t – 有索引指向本身
local tt = read_only(t)
print(tt.a, tt.b, tt.c)
print(tt.c)
print(tt.c.a, tt.c.b)
–输出
–1       2       table: 0039C6A8
–1       2
–如果不用 travelled_tables，将会进入死循环
]]
