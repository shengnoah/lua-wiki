    1 Logistic Regression 逻辑回归

    library(VGAM) # 广义线性模型


    #Build the model
    model1<-vglm(class~buying+maint+doors+persons+lug_boot+safety,family = &#34;multinomial&#34;,data=car_eval)
    

    #Summarize the model
    

    summary(model1)
    #Predict using the model
    x<-car_eval[,1:6]
    y<-car_eval[,7]
    probability<-predict(model1,x,type=&#34;response&#34;) # 相应就是二元
    car_eval$pred_log_reg<-apply(probability,1,which.max)
    car_eval$pred_log_reg[which(car_eval$pred_log_reg==&#34;1&#34;)]<-levels(car_eval$class)[1]
    car_eval$pred_log_reg[which(car_eval$pred_log_reg==&#34;2&#34;)]<-levels(car_eval$class)[2]
    car_eval$pred_log_reg[which(car_eval$pred_log_reg==&#34;3&#34;)]<-levels(car_eval$class)[3]
    car_eval$pred_log_reg[which(car_eval$pred_log_reg==&#34;4&#34;)]<-levels(car_eval$class)[4]
    

    #Accuracy of the model
    mtab<-table(car_eval$pred_log_reg,car_eval$class)
    library(caret)
    confusionMatrix(mtab)
